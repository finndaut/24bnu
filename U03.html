import numpy as np
import matplotlib.pyplot as plt

m_range = np.arange(1,100)
a = -4
b = 4


def error(m):
    i_range = np.arange(m + 1)
    x_range = a + (b-a) / m * i_range
    N = 2001
    t_range = a + (b-a) / N * np.arange(1, N)


    def newton(x_val):
        m =  len(x_range)
        s_range = np.exp(- x_range ** 2 / 2)
        b_list = np.zeros(shape=(m,m))
        b_list[0] = s_range
        ret = b_list[0][0]
        for i in range(1,m):
            b_list[i,i:m] = np.diff(b_list[i-1,i-1:m]) / (x_range[i:m]-x_range[0:m-i])
            ret += b_list[i][i] * np.prod(x_val-x_range[0:i])
        return ret
    

    newton = np.vectorize(newton)

    return np.linalg.norm(np.exp(- t_range ** 2 / 2) - newton(t_range), ord=1)


error = np.vectorize(error)

fig, ax = plt.subplots()
error_range = error(m_range)
ax.semilogy(m_range, error_range)
ax.set(xlabel='$m$',ylabel='$e_m$')
